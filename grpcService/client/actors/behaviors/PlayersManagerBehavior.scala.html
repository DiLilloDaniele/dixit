44 class PlayersManagerBehaviorImpl(context: ActorContext[Command],
45                                  maxPlayers: Int,
46                                  foreman: ActorRef[ForemanBehavior.Command]) extends AbstractBehavior[Command](context):
47 
48   var playersList: List[ActorRef[PlayerBehavior.Command]] = List()
49 
50   override def onMessage(msg: Command): Behavior[Command] = msg match {
51     case Startup => waitPlayers(maxPlayers, 0, foreman)
52     case _ => Behaviors.same
53   }
54 
55   def waitPlayers(maxPlayers: Int, currentNumPlayers: Int, foreman: ActorRef[ForemanBehavior.Command]): Behavior[Command] = Behaviors.receiveMessage[Command] {
56     case AskEnter(replyTo) =>
57       val currentPlayers = currentNumPlayers + 1
58       if (currentPlayers <= maxPlayers) {
59         playersList = playersList :+ replyTo
60         replyTo ! PlayerBehavior.MemberOK
61         if(currentPlayers == maxPlayers)
62           foreman ! ForemanBehavior.Start(playersList.toSet)
63           gameOn(maxPlayers, currentPlayers, foreman)
64         else
65           waitPlayers(maxPlayers, currentPlayers, foreman)
66       } else {
67         replyTo ! PlayerBehavior.MemberKO
68         Behaviors.same
69       }
70     case PlayerExited =>
71       waitPlayers(maxPlayers, currentNumPlayers - 1, foreman)
72     case _ =>
73       Behaviors.same
74   }
75 
76   def gameOn(maxPlayers: Int, currentNumPlayers: Int, foreman: ActorRef[ForemanBehavior.Command]): Behavior[Command] = Behaviors.receiveMessage[Command] {
77     case AskEnter(replyTo) =>
78       val playersAddresses = playersList.map { i => i.path.address.toString}
79       if(playersAddresses.contains(replyTo.path.address.toString))
80         context.log.info("un giocatore è rientrato")
81         replyTo ! PlayerBehavior.MemberOK
82         foreman ! ForemanBehavior.PlayerRejoined(replyTo.path.address.toString)
83         Behaviors.same
84       val currentPlayers = currentNumPlayers + 1
85       gameOn(maxPlayers, currentPlayers, foreman)
86     case PlayerExited =>
87       val currentPlayers = currentNumPlayers - 1
88       context.log.info("un giocatore ha quittato")
89       Behaviors.withTimers { timer =>
90         timer.startSingleTimer(CheckMembers, 15000 milliseconds)
91         gameOn(maxPlayers, currentPlayers, foreman)
92       }
93 
94     case CheckMembers =>
95       context.log.info("Checcko....")
96       if(currentNumPlayers < maxPlayers)
97         context.log.info("Stop game")
98         context.self ! Stop
99         Behaviors.same
100       else
101         context.log.info("NO stop game")
102         Behaviors.same
103 
104     case Stop =>
105       playersList.foreach { i =>
106         i ! PlayerBehavior.GameInfo("Uno o più giocatori si sono disconnessi. Partita annullata :(")
107       }
108       Behaviors.stopped
109   }
