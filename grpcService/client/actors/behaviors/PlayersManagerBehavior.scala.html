46 class PlayersManagerBehaviorImpl(context: ActorContext[Command],
47                                  maxPlayers: Int,

50   var playersList: List[ActorRef[PlayerBehavior.Command]] = List()
51 
52   override def onMessage(msg: Command): Behavior[Command] = msg match {
53     case Startup => waitPlayers(maxPlayers, 0, foreman)
54     case _ => Behaviors.same

57   def waitPlayers(maxPlayers: Int, currentNumPlayers: Int, foreman: ActorRef[ForemanBehavior.Command]): Behavior[Command] = Behaviors.receiveMessage[Command] {
58     case AskEnter(replyTo) =>
59       val currentPlayers = currentNumPlayers + 1
60       if (currentPlayers <= maxPlayers) {
61         playersList = playersList :+ replyTo
62         replyTo ! PlayerBehavior.MemberOK
63         if(currentPlayers == maxPlayers)
64           foreman ! ForemanBehavior.Start(playersList.toSet)
65           gameOn(maxPlayers, currentPlayers, foreman)
66         else
67           waitPlayers(maxPlayers, currentPlayers, foreman)
68       } else {
69         replyTo ! PlayerBehavior.MemberKO
70         Behaviors.same
71       }
72     case PlayerExited(address) =>
73       playersList = playersList filter { i => i.path.address != address }
74       context.log.info("PLAYER TOLTO: " + playersList)
75       waitPlayers(maxPlayers, currentNumPlayers - 1, foreman)
76     case _ =>
77       Behaviors.same
78   }
79 
80   def gameOn(maxPlayers: Int, currentNumPlayers: Int, foreman: ActorRef[ForemanBehavior.Command]): Behavior[Command] = Behaviors.receiveMessage[Command] {
81     case AskEnter(replyTo) =>
82       val playersAddresses = playersList.map { i => i.path.address.toString}
83       if(playersAddresses.contains(replyTo.path.address.toString))
84         context.log.info("un giocatore è rientrato")
85         replyTo ! PlayerBehavior.MemberOK
86         foreman ! ForemanBehavior.PlayerRejoined(replyTo.path.address.toString)
87         Behaviors.same
88         // TODO altrimenti invia KO
89       val currentPlayers = currentNumPlayers + 1
90       gameOn(maxPlayers, currentPlayers, foreman)
91     case PlayerExited(address) =>
92       val currentPlayers = currentNumPlayers - 1
93       context.log.info("un giocatore ha quittato")
94       Behaviors.withTimers { timer =>
95         timer.startSingleTimer(CheckMembers, 15000 milliseconds)
96         gameOn(maxPlayers, currentPlayers - 1, foreman)
97       }
98 
99     case CheckMembers =>
100       context.log.info("Checcko....")
101       if(currentNumPlayers < maxPlayers)
102         context.log.info("Stop game")
103         context.self ! Stop
104         Behaviors.same
105       else
106         context.log.info("NO stop game")
107         Behaviors.same
108 
109     case Stop =>
110       playersList.foreach { i =>
111         i ! PlayerBehavior.GameInfo("Uno o più giocatori si sono disconnessi. Partita annullata :(")
112       }
113       Behaviors.stopped
114   }
