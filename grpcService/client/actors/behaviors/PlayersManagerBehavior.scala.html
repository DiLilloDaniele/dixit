46 class PlayersManagerBehaviorImpl(context: ActorContext[Command],
47                                  maxPlayers: Int,
48                                  foreman: ActorRef[ForemanBehavior.Command]) extends AbstractBehavior[Command](context):
49 
50   var playersList: List[ActorRef[PlayerBehavior.Command]] = List()
51   var currentNumPlayers: Int = maxPlayers
52 
53   override def onMessage(msg: Command): Behavior[Command] = msg match {
54     case Startup => waitPlayers(0, foreman)
55     case _ => Behaviors.same
56   }
57 
58   def waitPlayers(currentNumPlayers: Int, foreman: ActorRef[ForemanBehavior.Command]): Behavior[Command] = Behaviors.receiveMessage[Command] {
59     case AskEnter(replyTo) =>
60       val currentPlayers = currentNumPlayers + 1
61       if (currentPlayers <= maxPlayers) {
62         playersList = playersList :+ replyTo
63         replyTo ! PlayerBehavior.MemberOK
64         if(currentPlayers == maxPlayers)
65           foreman ! ForemanBehavior.Start(playersList.toSet)
66           gameOn(foreman)
67         else
68           waitPlayers(currentPlayers, foreman)
69       } else {
70         replyTo ! PlayerBehavior.MemberKO
71         Behaviors.same
72       }
73     case PlayerExited(address) =>
74       playersList = playersList filter { i => i.path.address != address }
75       context.log.info("PLAYER TOLTO: " + playersList)
76       waitPlayers(currentNumPlayers - 1, foreman)
77     case _ =>
78       Behaviors.same
79   }
80 
81   def gameOn(foreman: ActorRef[ForemanBehavior.Command]): Behavior[Command] = Behaviors.receiveMessage[Command] {
82     case AskEnter(replyTo) =>
83       val playersAddresses = playersList.map { i => i.path.address.toString}
84       if(playersAddresses.contains(replyTo.path.address.toString) && currentNumPlayers < maxPlayers)
85         context.log.info("un giocatore è rientrato")
86         replyTo ! PlayerBehavior.MemberOK
87         foreman ! ForemanBehavior.PlayerRejoined(replyTo.path.address.toString)
88         currentNumPlayers = currentNumPlayers + 1
89         if(currentNumPlayers == maxPlayers)
90           foreman ! ForemanBehavior.RestartTurn
91         gameOn(foreman)
92       else
93         replyTo ! PlayerBehavior.MemberKO
94         Behaviors.same
95     case PlayerExited(address) =>
96       currentNumPlayers = currentNumPlayers - 1
97       context.log.info("un giocatore ha quittato")
98       foreman ! ForemanBehavior.InterruptTurn
99       Behaviors.withTimers { timer =>
100         timer.startSingleTimer(CheckMembers, 5000 milliseconds)
101         gameOn(foreman)
102       }
103 
104     case CheckMembers =>
105       context.log.info("Checcko....")
106       if(currentNumPlayers < maxPlayers)
107         context.log.info("Stop game")
108         context.self ! Stop
109         Behaviors.same
110       else
111         context.log.info("NO stop game")
112         Behaviors.same
113 
114     case Stop =>
115       playersList.foreach { i =>
116         i ! PlayerBehavior.GameInfo("Uno o più giocatori si sono disconnessi. Partita annullata :(")
117       }
118       Behaviors.stopped
119   }
