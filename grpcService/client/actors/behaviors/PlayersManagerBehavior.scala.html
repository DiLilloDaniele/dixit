44 class PlayersManagerBehaviorImpl(context: ActorContext[Command],
45                                  maxPlayers: Int,
46                                  foreman: ActorRef[ForemanBehavior.Command]) extends AbstractBehavior[Command](context):
47 
48   var playersList: List[ActorRef[PlayerBehavior.Command]] = List()
49 
50   override def onMessage(msg: Command): Behavior[Command] = msg match {
51     case Startup => waitPlayers(maxPlayers, 0, foreman)
52     case _ => Behaviors.same

55   def waitPlayers(maxPlayers: Int, currentNumPlayers: Int, foreman: ActorRef[ForemanBehavior.Command]): Behavior[Command] = Behaviors.receiveMessage[Command] {
56     case AskEnter(replyTo) =>
57       val currentPlayers = currentNumPlayers + 1
58       if (currentPlayers <= maxPlayers) {
59         playersList = playersList :+ replyTo
60         replyTo ! PlayerBehavior.MemberOK
61         if(currentPlayers == maxPlayers)
62           foreman ! ForemanBehavior.Start(playersList.toSet)
63           gameOn(maxPlayers, currentPlayers, foreman)
64         else
65           waitPlayers(maxPlayers, currentPlayers, foreman)
66       } else {
67         replyTo ! PlayerBehavior.MemberKO
68         Behaviors.same
69       }
70       //TODO non proprio correttissimo, togliere dalla lista replyTo e fare mailbox a priorità
71     case PlayerExited =>
72       waitPlayers(maxPlayers, currentNumPlayers - 1, foreman)
73     case _ =>
74       Behaviors.same
75   }

77   def gameOn(maxPlayers: Int, currentNumPlayers: Int, foreman: ActorRef[ForemanBehavior.Command]): Behavior[Command] = Behaviors.receiveMessage[Command] {
78     case AskEnter(replyTo) =>
79       val playersAddresses = playersList.map { i => i.path.address.toString}
80       if(playersAddresses.contains(replyTo.path.address.toString))
81         context.log.info("un giocatore è rientrato")
82         replyTo ! PlayerBehavior.MemberOK
83         foreman ! ForemanBehavior.PlayerRejoined(replyTo.path.address.toString)
84         Behaviors.same
85         // TODO altrimenti invia KO
86       val currentPlayers = currentNumPlayers + 1
87       gameOn(maxPlayers, currentPlayers, foreman)
88     case PlayerExited =>
89       val currentPlayers = currentNumPlayers - 1
90       context.log.info("un giocatore ha quittato")
91       Behaviors.withTimers { timer =>
92         timer.startSingleTimer(CheckMembers, 15000 milliseconds)
93         gameOn(maxPlayers, currentPlayers - 1, foreman)
94       }
95 
96     case CheckMembers =>
97       context.log.info("Checcko....")
98       if(currentNumPlayers < maxPlayers)
99         context.log.info("Stop game")
100         context.self ! Stop
101         Behaviors.same
102       else
103         context.log.info("NO stop game")
104         Behaviors.same
105 
106     case Stop =>
107       playersList.foreach { i =>
108         i ! PlayerBehavior.GameInfo("Uno o più giocatori si sono disconnessi. Partita annullata :(")
109       }
110       Behaviors.stopped
111   }
