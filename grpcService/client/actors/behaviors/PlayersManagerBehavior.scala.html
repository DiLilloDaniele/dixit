46 class PlayersManagerBehaviorImpl(context: ActorContext[Command],
47                                  maxPlayers: Int,
48                                  foreman: ActorRef[ForemanBehavior.Command],
49                                  timeoutMillis: Int) extends AbstractBehavior[Command](context):
50 
51   var playersList: List[ActorRef[PlayerBehavior.Command]] = List()
52   var currentNumPlayers: Int = maxPlayers
53 
54   override def onMessage(msg: Command): Behavior[Command] = msg match {
55     case Startup => waitPlayers(0, foreman)
56     case _ => Behaviors.same
57   }
58 
59   def waitPlayers(currentNumPlayers: Int, foreman: ActorRef[ForemanBehavior.Command]): Behavior[Command] = Behaviors.receiveMessage[Command] {
60     case AskEnter(replyTo) =>
61       val currentPlayers = currentNumPlayers + 1
62       if (currentPlayers <= maxPlayers) {
63         playersList = playersList :+ replyTo
64         replyTo ! PlayerBehavior.MemberOK
65         if(currentPlayers == maxPlayers)
66           foreman ! ForemanBehavior.Start(playersList.toSet)
67           gameOn(foreman)
68         else
69           waitPlayers(currentPlayers, foreman)
70       } else {
71         replyTo ! PlayerBehavior.MemberKO
72         Behaviors.same
73       }
74     case PlayerExited(address) =>
75       playersList = playersList filter { i => i.path.address != address }
76       context.log.info("PLAYER TOLTO: " + playersList)
77       waitPlayers(currentNumPlayers - 1, foreman)
78     case _ =>
79       Behaviors.same
80   }
81 
82   def gameOn(foreman: ActorRef[ForemanBehavior.Command]): Behavior[Command] = Behaviors.receiveMessage[Command] {
83     case AskEnter(replyTo) =>
84       val playersAddresses = playersList.map { i => i.path.address.toString}
85       if(playersAddresses.contains(replyTo.path.address.toString) && currentNumPlayers < maxPlayers)
86         context.log.info("un giocatore è rientrato")
87         replyTo ! PlayerBehavior.MemberOK
88         foreman ! ForemanBehavior.PlayerRejoined(replyTo.path.address.toString)
89         currentNumPlayers = currentNumPlayers + 1
90         if(currentNumPlayers == maxPlayers)
91           foreman ! ForemanBehavior.RestartTurn
92         gameOn(foreman)
93       else
94         replyTo ! PlayerBehavior.MemberKO
95         Behaviors.same
96     case PlayerExited(address) =>
97       currentNumPlayers = currentNumPlayers - 1
98       context.log.info("un giocatore ha quittato")
99       foreman ! ForemanBehavior.InterruptTurn
100       Behaviors.withTimers { timer =>
101         timer.startSingleTimer(CheckMembers, timeoutMillis milliseconds)
102         gameOn(foreman)
103       }
104 
105     case CheckMembers =>
106       context.log.info("Checcko....")
107       if(currentNumPlayers < maxPlayers)
108         context.log.info("Stop game")
109         context.self ! Stop
110         Behaviors.same
111       else
112         context.log.info("NO stop game")
113         Behaviors.same
114 
115     case Stop =>
116       playersList.foreach { i =>
117         i ! PlayerBehavior.GameInfo("Uno o più giocatori si sono disconnessi. Partita annullata :(")
118       }
119       Behaviors.stopped
120   }
