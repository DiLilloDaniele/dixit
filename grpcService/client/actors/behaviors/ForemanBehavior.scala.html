159   def waitForSelections(actor: ActorRef[PlayerBehavior.Command], turn: Int, cardSelected: String): Behavior[Command | Receptionist.Listing] = Behaviors.receiveMessage {
160     case SelectionToApply(cardId, replyTo) =>
161       context.log.info("UNA SCELTA RICEVUTA")
162       playersSelection = playersSelection :+ PlayerSelection(cardId, replyTo)
163       // rimuovo la carta scelta dal mazzo del giocatore che l'ha scelta
164       playersList foreach { i =>
165         if(i.player.path == replyTo.path)
166           i.cards -= cardId
167       }
168 
169       if(playersSelection.size < (playersList.size - 1))
170         Behaviors.same
171       else
172         context.log.info("RICEVUTO LE CARTE DA TUTTI")
173         val playersSubmissions = playersSelection.map { i => i.card }.toList :+ cardSelected
174         playersList foreach { i =>
175           if(i.player.path != actor.path)
176             i.player ! CardsSubmittedByOthers(playersSubmissions)
177         }
178         waitForGuessing(actor, turn, cardSelected)
179 
180     case InterruptTurn =>
181       playersList foreach { i =>
182         i.player ! TurnCancelled
183       }
184       Behaviors.same
185 
186     case PlayerRejoined(address) =>
187       playerRejoinedProcedure(address, MAX_CARDS)
188       Behaviors.same
189     
190     case RestartTurn =>
191       restartTurnProcedure(actor)
192       manageTurnOf(actor, turn)
193 
194     case Stop => Behaviors.stopped
195 
196     case _ => Behaviors.same

199   def waitForGuessing(actor: ActorRef[PlayerBehavior.Command], turn: Int, cardSelected: String): Behavior[Command | Receptionist.Listing] = Behaviors.receiveMessage {
200     case GuessSelection(cardId, replyTo) =>
201       context.log.info("UNA GUESS RICEVUTA")
202       playersGuessing = playersGuessing :+ PlayerSelection(cardId, replyTo)
203       if(playersGuessing.size < (playersList.size - 1))
204         Behaviors.same
205       else
206         context.log.info("GUESSED LE CARTE DA TUTTI")
207         executeResults(playersGuessing, cardSelected, actor)
208 
209         val newTurn = turn match
210           case m if m == (playersList.size - 1) => 0
211           case _ => turn + 1
212 
213         if(turnsFinished >= (maxTurns - 1) || playersList.map { i => i.score }.contains(30))
214           context.log.info("FINE GIOCO")
215           playersList foreach { i =>
216             i.player ! PlayerBehavior.EndGame(i.score)
217           }
218           Behaviors.stopped
219         else
220           Behaviors.withTimers { timers =>
221             timers.startSingleTimer(NewTurn(newTurn), 5000 milliseconds)
222             Behaviors.same
223           }
224 
225     case NewTurn(turnIndex) =>
226       turnsFinished = turnsFinished + 1
227       // clean playersGuessing choices
228       playersGuessing = List()
229       playersList foreach { i =>
230         val newCard: String = pickCards(1).head.toString
231         i.cards += newCard
232         i.player ! PlayerBehavior.NewCard(newCard)
233       }
234       playersList(turnIndex).player ! YourTurn(root)
235       manageTurnOf(playersList(turnIndex).player, turnIndex)
236 
237     case InterruptTurn =>
238       playersList.foreach { i =>
239         i.player ! TurnCancelled
240       }
241       Behaviors.same
242 
243     case PlayerRejoined(address) =>
244       playerRejoinedProcedure(address, MAX_CARDS)
245       Behaviors.same
246     
247     case RestartTurn =>
248       restartTurnProcedure(actor)
249       manageTurnOf(actor, turn)
250 
251     case Stop => Behaviors.stopped
252       
253     case _ => Behaviors.same
254   }
