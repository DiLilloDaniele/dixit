157   def waitForSelections(actor: ActorRef[PlayerBehavior.Command], turn: Int, cardSelected: String): Behavior[Command | Receptionist.Listing] = Behaviors.receiveMessage {
158     case SelectionToApply(cardId, replyTo) =>
159       context.log.info("UNA SCELTA RICEVUTA")
160       playersSelection = playersSelection :+ PlayerSelection(cardId, replyTo)
161       // remove the chosen card from the narrator's hand
162       playersList foreach { i =>
163         if(i.player.path == replyTo.path)
164           i.cards -= cardId
165       }
166 
167       if(playersSelection.size < (playersList.size - 1))
168         Behaviors.same
169       else
170         context.log.info("RICEVUTO LE CARTE DA TUTTI")
171         val playersSubmissions = playersSelection.map { i => i.card }.toList :+ cardSelected
172         playersList foreach { i =>
173           if(i.player.path != actor.path)

175         }
176         waitForGuessing(actor, turn, cardSelected)

178     case InterruptTurn =>
179       playersList foreach { i =>
180         i.player ! TurnCancelled
181       }

185       playerRejoinedProcedure(address, MAX_CARDS)
186       Behaviors.same

197   def waitForGuessing(actor: ActorRef[PlayerBehavior.Command], turn: Int, cardSelected: String): Behavior[Command | Receptionist.Listing] = Behaviors.receiveMessage {
198     case GuessSelection(cardId, replyTo) =>
199       context.log.info("UNA GUESS RICEVUTA")
200       playersGuessing = playersGuessing :+ PlayerSelection(cardId, replyTo)
201       if(playersGuessing.size < (playersList.size - 1))
202         Behaviors.same
203       else
204         context.log.info("GUESSED LE CARTE DA TUTTI")
205         executeResults(playersGuessing, cardSelected, actor)
206 
207         val newTurn = turn match
208           case m if m == (playersList.size - 1) => 0
209           case _ => turn + 1
210 
211         if(turnsFinished >= (maxTurns - 1) || playersList.map { i => i.score }.contains(30))
212           context.log.info("FINE GIOCO")
213           playersList foreach { i =>
214             i.player ! PlayerBehavior.EndGame(i.score)
215           }
216           Behaviors.stopped
217         else
218           Behaviors.withTimers { timers =>
219             timers.startSingleTimer(NewTurn(newTurn), 5000 milliseconds)
220             Behaviors.same
221           }
222 
223     case NewTurn(turnIndex) =>
224       turnsFinished = turnsFinished + 1
225       // clean playersGuessing choices
226       playersGuessing = List()
227       playersList foreach { i =>
228         val newCard: String = pickCards(1).head.toString
229         i.cards += newCard
230         i.player ! PlayerBehavior.NewCard(newCard)
231       }
232       playersList(turnIndex).player ! YourTurn(root)
233       manageTurnOf(playersList(turnIndex).player, turnIndex)
234 
235     case InterruptTurn =>
236       playersList.foreach { i =>
237         i.player ! TurnCancelled
238       }
239       Behaviors.same
240 
241     case PlayerRejoined(address) =>
242       playerRejoinedProcedure(address, MAX_CARDS)
243       Behaviors.same
244     
245     case RestartTurn =>
246       restartTurnProcedure(actor)
247       manageTurnOf(actor, turn)
248 
249     case Stop => Behaviors.stopped
250       
251     case _ => Behaviors.same
252   }
