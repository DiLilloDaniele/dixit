46     Behaviors.setup[Command] { ctx =>

86   override def onMessage(msg: Command | Receptionist.Listing): Behavior[Command | Receptionist.Listing] = msg match {
87     case msg: Receptionist.Listing =>
88       context.log.info("RECEIVED SOMETHING")
89       val services: List[ActorRef[PlayersManagerBehavior.Command]] = msg.serviceInstances(PlayersManagerBehavior.Service).toList
90       if(!services.isEmpty)
91         context.self ! AskToEnter(services.head)
92       Behaviors.same
93     case Start =>
94       context.system.receptionist ! Receptionist.Subscribe(PlayersManagerBehavior.Service, context.self)
95       Behaviors.same
96     case AskToEnter(replyTo) =>
97       replyTo ! PlayersManagerBehavior.AskEnter(rootActor)
98       Behaviors.same
99     case MemberOK =>
100       context.log.info("MEMBER OK")
101       waitForCards()
102     case MemberKO =>
103       interactionActor ! InteractionBehavior.MessageError("Impossibile entrare")
104       Behaviors.stopped
105     case m =>
106       context.log.info("Unknown message: " + m.toString)
107       Behaviors.same
108   }

119   def gameOn(): Behavior[Command | Receptionist.Listing] = Behaviors.receiveMessage {
120     case YourTurn(replyTo) => 
121       // deve inviare un CardToGuess ad un attore InteractionActor, che può essere 
122       // un attore che sceglie random una carta e la invia qui, o la gui vera e propria
123       context.log.info("SONO IL PLAYER, E' IL MIO TURNO")
124       interactionActor ! InteractionBehavior.ChooseTheCard(context.self)
125       choose(replyTo)
126     case CardChoosenByOther(title, replyTo) =>
127       interactionActor ! InteractionBehavior.GuessCard(rootActor, title)
128       guess(title, replyTo)
129     case NewCard(card) =>
130       interactionActor ! InteractionBehavior.NewCard(card)
131       Behaviors.same
132     case TurnCancelled => Behaviors.same
133 
134     case EndGame(points) =>
135       interactionActor ! InteractionBehavior.MessageError(s"Gioco terminato, punti guadagnati: $points")
136       onStop(points)
137       Behaviors.stopped
138 
139     case _ => Behaviors.same
140   }

142   def guess(title: String, replyTo: ActorRef[ForemanBehavior.Command]): Behavior[Command | Receptionist.Listing] = Behaviors.receiveMessage {
143     case SingleCardSubmitted(card) =>
144       replyTo ! ForemanBehavior.SelectionToApply(card, rootActor)
145       Behaviors.same
146     case CardsSubmittedByOthers(cards) =>
147       interactionActor ! InteractionBehavior.ShowCardsProposed(rootActor, cards, title)
148       choose(replyTo)
149     case TurnCancelled => gameOn()
150 
151     case _ => Behaviors.same
152   }

154   def choose(replyTo: ActorRef[ForemanBehavior.Command]): Behavior[Command | Receptionist.Listing] = Behaviors.receiveMessage {
155     case CardChoosenForTurn(cardId, title) =>
156       replyTo ! ForemanBehavior.CardToGuess(cardId, title, rootActor)
157       interactionActor ! InteractionBehavior.ResetInteraction
158       gameOn()
159     case GuessedCardbyUser(cardId) =>
160       replyTo ! ForemanBehavior.GuessSelection(cardId, rootActor)
161       waitForCardRevealed(replyTo)
162 
163     case TurnCancelled => gameOn()
164 
165     case msg => context.log.info("Unknown message: " + msg)
166       Behaviors.same
167   }
