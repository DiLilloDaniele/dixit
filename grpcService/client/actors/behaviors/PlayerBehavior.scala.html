84   override def onMessage(msg: Command | Receptionist.Listing): Behavior[Command | Receptionist.Listing] = msg match {
85     case msg: Receptionist.Listing =>
86       context.log.info("RECEIVED SOMETHING")
87       val services: List[ActorRef[PlayersManagerBehavior.Command]] = msg.serviceInstances(PlayersManagerBehavior.Service).toList
88       if(!services.isEmpty)
89         context.self ! AskToEnter(services.head)
90       Behaviors.same
91     case Start =>
92       context.system.receptionist ! Receptionist.Subscribe(PlayersManagerBehavior.Service, context.self)
93       Behaviors.same
94     case AskToEnter(replyTo) =>
95       replyTo ! PlayersManagerBehavior.AskEnter(rootActor)
96       Behaviors.same
97     case MemberOK =>
98       context.log.info("MEMBER OK")
99       waitForCards()
100     case MemberKO =>
101       interactionActor ! InteractionBehavior.MessageError("Impossibile entrare")
102       Behaviors.stopped
103     case m =>
104       context.log.info("Unknown message: " + m.toString)
105       Behaviors.same
106   }

117   def gameOn(): Behavior[Command | Receptionist.Listing] = Behaviors.receiveMessage {
118     case YourTurn(replyTo) =>
119       context.log.info("SONO IL PLAYER, E' IL MIO TURNO")
120       interactionActor ! InteractionBehavior.ChooseTheCard(context.self)
121       choose(replyTo)
122     case CardChoosenByOther(title, replyTo) =>
123       interactionActor ! InteractionBehavior.GuessCard(rootActor, title)
124       guess(title, replyTo)
125     case NewCard(card) =>
126       interactionActor ! InteractionBehavior.NewCard(card)
127       Behaviors.same
128     case TurnCancelled => 
129       interactionActor ! InteractionBehavior.MessageError("Turno cancellato, attendere...")
130       interactionActor ! InteractionBehavior.ResetInteraction
131       Behaviors.same
132 
133     case EndGame(points) =>
134       interactionActor ! InteractionBehavior.EndGame(s"Gioco terminato, punti guadagnati: $points")
135       onStop(points)
136       Behaviors.stopped
137 
138     case _ => Behaviors.same
139   }
