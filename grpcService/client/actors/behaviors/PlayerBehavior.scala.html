45   def apply(logger: Option[ActorRef[Command]] = Option.empty, interactionExt: Option[ActorRef[InteractionBehavior.Command]] = Option.empty): Behavior[Command] =
46     Behaviors.setup[Command] { ctx =>
47       ctx.system.receptionist ! Receptionist.Register(Service, ctx.self)
48 
49       val interaction = interactionExt match {
50         case Some(ref) =>

75           }

79         ctx.log.info("System terminated [PLAYER]. Shutting down")

86   override def onMessage(msg: Command | Receptionist.Listing): Behavior[Command | Receptionist.Listing] = msg match {
87     case msg: Receptionist.Listing =>
88       context.log.info("RECEIVED SOMETHING")
89       val services: List[ActorRef[PlayersManagerBehavior.Command]] = msg.serviceInstances(PlayersManagerBehavior.Service).toList
90       if(!services.isEmpty)
91         context.self ! AskToEnter(services.head)
92       Behaviors.same
93     case Start =>
94       context.system.receptionist ! Receptionist.Subscribe(PlayersManagerBehavior.Service, context.self)
95       Behaviors.same
96     case AskToEnter(replyTo) =>
97       replyTo ! PlayersManagerBehavior.AskEnter(rootActor)
98       Behaviors.same
99     case MemberOK =>
100       context.log.info("MEMBER OK")
101       waitForCards()
102     case MemberKO =>
103       interactionActor ! InteractionBehavior.MessageError("Impossibile entrare")
104       Behaviors.stopped
105     case m =>
106       context.log.info("Unknown message: " + m.toString)
107       Behaviors.same
108   }

119   def gameOn(): Behavior[Command | Receptionist.Listing] = Behaviors.receiveMessage {
120     case YourTurn(replyTo) =>

123       context.log.info("SONO IL PLAYER, E' IL MIO TURNO")
124       interactionActor ! InteractionBehavior.ChooseTheCard(context.self)

127       interactionActor ! InteractionBehavior.GuessCard(rootActor, title)

129     case NewCard(card) =>
130       interactionActor ! InteractionBehavior.NewCard(card)

132     case TurnCancelled => Behaviors.same
133 
134     case EndGame(points) =>
135       interactionActor ! InteractionBehavior.MessageError(s"Gioco terminato, punti guadagnati: $points")
136       Behaviors.stopped

139   }

141   def guess(title: String, replyTo: ActorRef[ForemanBehavior.Command]): Behavior[Command | Receptionist.Listing] = Behaviors.receiveMessage {
142     case SingleCardSubmitted(card) =>
143       replyTo ! ForemanBehavior.SelectionToApply(card, rootActor)
144       Behaviors.same
145     case CardsSubmittedByOthers(cards) =>
146       interactionActor ! InteractionBehavior.ShowCardsProposed(rootActor, cards, title)
147       choose(replyTo)
148     case TurnCancelled => gameOn()
149 
150     case _ => Behaviors.same
151   }

153   def choose(replyTo: ActorRef[ForemanBehavior.Command]): Behavior[Command | Receptionist.Listing] = Behaviors.receiveMessage {
154     case CardChoosenForTurn(cardId, title) =>
155       replyTo ! ForemanBehavior.CardToGuess(cardId, title, rootActor)
156       interactionActor ! InteractionBehavior.ResetInteraction
157       gameOn()
158     case GuessedCardbyUser(cardId) =>
159       replyTo ! ForemanBehavior.GuessSelection(cardId, rootActor)
160       waitForCardRevealed(replyTo)
161 
162     case TurnCancelled => gameOn()
163 
164     case msg => context.log.info("Unknown message: " + msg)
165       Behaviors.same
166   }
