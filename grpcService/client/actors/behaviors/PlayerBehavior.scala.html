43   def apply(logger: Option[ActorRef[Command]] = Option.empty, interactionExt: Option[ActorRef[InteractionBehavior.Command]] = Option.empty, onStop: (Int) => Unit = (v) => ()): Behavior[Command] =
44     Behaviors.setup[Command] { ctx =>
45       ctx.system.receptionist ! Receptionist.Register(Service, ctx.self)
46 
47       val interaction = interactionExt match {
48         case Some(ref) =>
49           ctx.log.info("INTERACTION ESTERNO")
50           ref
51         case _ =>
52           ctx.log.info("INTERACTION INTERNO")
53           ctx.spawn(Behaviors.setup[InteractionBehavior.Command](ctx1 => InteractionGuiImpl(ctx1, ctx.self)), "gui")
54       }
55       val player = ctx.spawn(Behaviors.setup[Command | Receptionist.Listing](context => new PlayerBehaviorImpl(context, interaction, ctx.self, onStop)), "player")
56       val listener = ctx.spawn(Behaviors.setup[Event | Receptionist.Listing](context => PlayerClusterListener(ctx.self)), "player-cluster-listener")

58       ctx.watch(interaction)
59 
60       Behaviors.receiveMessage[Command] {
61         case GameInfo(msg) =>

66           }

73           }

77         ctx.log.info("System terminated [PLAYER]. Shutting down")

84   override def onMessage(msg: Command | Receptionist.Listing): Behavior[Command | Receptionist.Listing] = msg match {
85     case msg: Receptionist.Listing =>
86       context.log.info("RECEIVED SOMETHING")
87       val services: List[ActorRef[PlayersManagerBehavior.Command]] = msg.serviceInstances(PlayersManagerBehavior.Service).toList
88       if(!services.isEmpty)
89         context.self ! AskToEnter(services.head)
90       Behaviors.same
91     case Start =>
92       context.system.receptionist ! Receptionist.Subscribe(PlayersManagerBehavior.Service, context.self)
93       Behaviors.same
94     case AskToEnter(replyTo) =>
95       replyTo ! PlayersManagerBehavior.AskEnter(rootActor)
96       Behaviors.same
97     case MemberOK =>
98       context.log.info("MEMBER OK")
99       waitForCards()
100     case MemberKO =>
101       interactionActor ! InteractionBehavior.MessageError("Impossibile entrare")
102       Behaviors.stopped
103     case m =>
104       context.log.info("Unknown message: " + m.toString)
105       Behaviors.same
106   }

117   def gameOn(): Behavior[Command | Receptionist.Listing] = Behaviors.receiveMessage {
118     case YourTurn(replyTo) => 
119       // deve inviare un CardToGuess ad un attore InteractionActor, che può essere 
120       // un attore che sceglie random una carta e la invia qui, o la gui vera e propria
121       context.log.info("SONO IL PLAYER, E' IL MIO TURNO")
122       interactionActor ! InteractionBehavior.ChooseTheCard(context.self)
123       choose(replyTo)
124     case CardChoosenByOther(title, replyTo) =>
125     //TODO rename to TitleChoosenByOther
126       interactionActor ! InteractionBehavior.GuessCard(rootActor, title)
127       guess(title, replyTo)
128     case NewCard(card) =>
129       interactionActor ! InteractionBehavior.NewCard(card)
130       Behaviors.same
131     case TurnCancelled => Behaviors.same
132 
133     case EndGame(points) =>
134       interactionActor ! InteractionBehavior.MessageError(s"Gioco terminato, punti guadagnati: $points")
135       onStop(points)
136       Behaviors.stopped
137 
138     case _ => Behaviors.same
139   }

141   def guess(title: String, replyTo: ActorRef[ForemanBehavior.Command]): Behavior[Command | Receptionist.Listing] = Behaviors.receiveMessage {
142     case SingleCardSubmitted(card) =>
143       replyTo ! ForemanBehavior.SelectionToApply(card, rootActor)
144       Behaviors.same
145     case CardsSubmittedByOthers(cards) =>
146       interactionActor ! InteractionBehavior.ShowCardsProposed(rootActor, cards, title)
147       choose(replyTo)
148     case TurnCancelled => gameOn()
149 
150     case _ => Behaviors.same
151   }

153   def choose(replyTo: ActorRef[ForemanBehavior.Command]): Behavior[Command | Receptionist.Listing] = Behaviors.receiveMessage {
154     case CardChoosenForTurn(cardId, title) =>
155       replyTo ! ForemanBehavior.CardToGuess(cardId, title, rootActor)
156       interactionActor ! InteractionBehavior.ResetInteraction
157       gameOn()
158     case GuessedCardbyUser(cardId) =>
159       replyTo ! ForemanBehavior.GuessSelection(cardId, rootActor)
160       waitForCardRevealed(replyTo)
161 
162     case TurnCancelled => gameOn()
163 
164     case msg => context.log.info("Unknown message: " + msg)
165       Behaviors.same
166   }
