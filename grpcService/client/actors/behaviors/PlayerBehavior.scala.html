85   override def onMessage(msg: Command | Receptionist.Listing): Behavior[Command | Receptionist.Listing] = msg match {
86     case msg: Receptionist.Listing =>
87       context.log.info("RECEIVED SOMETHING")
88       val services: List[ActorRef[PlayersManagerBehavior.Command]] = msg.serviceInstances(PlayersManagerBehavior.Service).toList
89       if(!services.isEmpty)
90         context.self ! AskToEnter(services.head)
91       Behaviors.same
92     case Start =>
93       context.system.receptionist ! Receptionist.Subscribe(PlayersManagerBehavior.Service, context.self)
94       Behaviors.same
95     case AskToEnter(replyTo) =>
96       replyTo ! PlayersManagerBehavior.AskEnter(rootActor)
97       Behaviors.same
98     case MemberOK =>
99       context.log.info("MEMBER OK")
100       waitForCards()
101     case MemberKO =>
102       interactionActor ! InteractionBehavior.MessageError("Impossibile entrare")
103       Behaviors.stopped
104     case m =>
105       context.log.info("Unknown message: " + m.toString)
106       Behaviors.same
107   }

118   def gameOn(): Behavior[Command | Receptionist.Listing] = Behaviors.receiveMessage {
119     case YourTurn(replyTo) =>
120       context.log.info("SONO IL PLAYER, E' IL MIO TURNO")
121       interactionActor ! InteractionBehavior.ChooseTheCard(context.self)
122       choose(replyTo)
123     case CardChoosenByOther(title, replyTo) =>
124     //TODO rename to TitleChoosenByOther
125       interactionActor ! InteractionBehavior.GuessCard(rootActor, title)
126       guess(title, replyTo)
127     case NewCard(card) =>
128       interactionActor ! InteractionBehavior.NewCard(card)
129       Behaviors.same
130     case TurnCancelled => 
131       interactionActor ! InteractionBehavior.MessageError("Turno cancellato, attendere...")
132       interactionActor ! InteractionBehavior.ResetInteraction
133       Behaviors.same
134 
135     case EndGame(points) =>
136       interactionActor ! InteractionBehavior.EndGame(s"Gioco terminato, punti guadagnati: $points")
137       onStop(points)
138       Behaviors.stopped
139 
140     case _ => Behaviors.same
141   }
