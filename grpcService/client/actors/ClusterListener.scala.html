34   def apply(manager: ActorRef[PlayersManagerBehavior.Command]): Behavior[Command | Event | Receptionist.Listing] = Behaviors.setup { ctx =>

36     // MemberEvent extends ClusterDomainEvent
37     // We use the "message adapter" pattern to avoid the need of directly supporting the whole MemberEvent messaging interface
38     val memberEventAdapter: ActorRef[MemberEvent] = ctx.messageAdapter(MemberChange.apply)
39     // To subscribe, you must provide your ActorRef[A<:ClusterDomainEvent] and the ClusterDomainEvent class
40     Cluster(ctx.system).subscriptions ! Subscribe(memberEventAdapter, classOf[MemberEvent])
41 
42     // ReachabilityEvent also extends ClusterDomainEvent
43     val reachabilityAdapter: ActorRef[ReachabilityEvent] = ctx.messageAdapter(ReachabilityChange.apply)
44     Cluster(ctx.system).subscriptions ! Subscribe(reachabilityAdapter, classOf[ReachabilityEvent])
45 
46     // Our behaviour listens for cluster events (membership and reachability events)
47     Behaviors.receiveMessage { message =>
48       message match {
49 
50         case ReachabilityChange(reachabilityEvent) =>
51           reachabilityEvent match {
52             case UnreachableMember(member) =>
53               ctx.log.info("Member detected as unreachable: {}", member)
54             case ReachableMember(member) =>
55               ctx.log.info("Member back to reachable: {}", member)
56           }
57 
58         case MemberChange(changeEvent) =>
59           changeEvent match {
60             case MemberUp(member) =>
61               ctx.log.info("Member is Up: {} - I'm {}", member.address, ctx.system.address.port)
62             case MemberRemoved(member, previousStatus) =>
63               ctx.log.info(s"ROLE MEMBER: ${member.roles.contains("player")}")
64               if(member.roles.contains("player"))
65                 ctx.log.info("OUT: {}", member.roles)
66                 ctx.log.info("Member is Removed: {} after {}", member.address, previousStatus)
67                 manager ! PlayersManagerBehavior.PlayerExited(member.address)
68             case _: MemberEvent => // ignore
69           }
70         case _ => Behaviors.same
